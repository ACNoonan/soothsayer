% Pandoc template for the Soothsayer Paper 1 build pipeline.
%
% Wraps Pandoc's body output in the arxiv-style preamble (arxiv.sty,
% NIPS-derived single-column) plus a title/author block, natbib
% bibliography, and the standard math + graphicx packages.
%
% NOTE: Pandoc template variables (single-dollar-delimited) are
% deliberately NOT referenced in these comments because Pandoc would
% substitute them verbatim and corrupt the preamble. Body/title/author/
% date come from the body markdown + the explicit \title{} \author{}
% \date{} below.
\documentclass[11pt]{article}

\usepackage{arxiv}

\usepackage[utf8]{inputenc}
\usepackage[T1]{fontenc}

% pdflatex DROPS raw-Unicode glyphs it has no declaration for, emitting
% only a log error -- the 2026-05 builds shipped blank check/ballot-x
% cells in the gate tables and a missing Delta in table headers.
% Declare every non-ASCII glyph the markdown sources use in text mode.
\usepackage{pifont}% \ding{51}/\ding{55} check / ballot-x
\DeclareUnicodeCharacter{0394}{\ensuremath{\Delta}}
\DeclareUnicodeCharacter{03A3}{\ensuremath{\Sigma}}
\DeclareUnicodeCharacter{03B4}{\ensuremath{\delta}}
\DeclareUnicodeCharacter{03BB}{\ensuremath{\lambda}}
\DeclareUnicodeCharacter{03C1}{\ensuremath{\rho}}
\DeclareUnicodeCharacter{03C3}{\ensuremath{\sigma}}
\DeclareUnicodeCharacter{03C4}{\ensuremath{\tau}}
\DeclareUnicodeCharacter{2194}{\ensuremath{\leftrightarrow}}
\DeclareUnicodeCharacter{2208}{\ensuremath{\in}}
\DeclareUnicodeCharacter{2212}{\ensuremath{-}}
\DeclareUnicodeCharacter{222A}{\ensuremath{\cup}}
\DeclareUnicodeCharacter{2248}{\ensuremath{\approx}}
\DeclareUnicodeCharacter{2260}{\ensuremath{\neq}}
\DeclareUnicodeCharacter{2261}{\ensuremath{\equiv}}
\DeclareUnicodeCharacter{2264}{\ensuremath{\leq}}
\DeclareUnicodeCharacter{2265}{\ensuremath{\geq}}
\DeclareUnicodeCharacter{2308}{\ensuremath{\lceil}}
\DeclareUnicodeCharacter{2309}{\ensuremath{\rceil}}
\DeclareUnicodeCharacter{2713}{\ding{51}}
\DeclareUnicodeCharacter{2717}{\ding{55}}

\providecommand{\tightlist}{%
  \setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}}

% Pandoc wraps \includegraphics output with \pandocbounded; clamp the
% boxed content to the surrounding column so oversized figures (e.g. the
% 12-inch native simulation_summary.pdf) do not bleed off the right margin.
\providecommand{\pandocbounded}[1]{%
  \resizebox{\ifdim\width>\linewidth\linewidth\else\width\fi}{!}{#1}}
